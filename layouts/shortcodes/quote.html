{{- /* ABOUTME: Pull-quote shortcode. Back-compat `attribution=` path + structured testimonial path. */ -}} {{- /* ABOUTME: Structured fields: name, role, organization, photo, featured. See docs/shortcodes.md. */ -}} {{- /* Resolution chain — when `attribution` is set, all structured fields are ignored (back-compat). */ -}} {{- $attribution := .Get "attribution" -}} {{- $name := .Get "name" -}} {{- $role := .Get "role" -}} {{- $organization := .Get "organization" -}} {{- $photo := .Get "photo" -}} {{- $featured := .Get "featured" -}} {{- /* Resolve photo URL using the same pattern as video-embed.html's poster_image: absolute URL (http/https prefix) -> verbatim site-root path (leading /) -> verbatim (Hugo emits as-is) relative filename -> page resource via .Page.Resources.GetMatch Only consulted when attribution is empty (back-compat path ignores it). */ -}} {{- $photoURL := "" -}} {{- if and (not $attribution) $photo -}} {{- if or (hasPrefix $photo "http://") (hasPrefix $photo "https://") -}} {{- $photoURL = $photo -}} {{- else if hasPrefix $photo "/" -}} {{- $photoURL = $photo -}} {{- else -}} {{- $res := .Page.Resources.GetMatch $photo -}} {{- if $res -}} {{- $photoURL = $res.RelPermalink -}} {{- else -}} {{- warnf "quote shortcode on %s: photo %q cannot be resolved as page resource, site-root path, or absolute URL" .Page.Path $photo -}} {{- end -}} {{- end -}} {{- end -}} {{- /* Build the "role line" — joins role + organization with ", " when both are set; otherwise renders whichever single one is set. */ -}} {{- $roleLine := "" -}} {{- if and $role $organization -}} {{- $roleLine = printf "%s, %s" $role $organization -}} {{- else if $role -}} {{- $roleLine = $role -}} {{- else if $organization -}} {{- $roleLine = $organization -}} {{- end -}} {{- /* Featured modifier class on the outer figure. Hugo's .Get returns a bool when the shortcode is called with `featured=true` (bare) and a string when called with `featured="true"`. Accept either. */ -}} {{- $isFeatured := false -}} {{- if or (eq $featured true) (eq $featured "true") -}} {{- $isFeatured = true -}} {{- end -}} {{- $figureClass := "pull-quote" -}} {{- if and (not $attribution) $isFeatured -}} {{- $figureClass = "pull-quote pull-quote-featured" -}} {{- end -}}
