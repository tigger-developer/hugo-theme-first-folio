{{/* ABOUTME: Generates wash CSS custom properties from hugo.yaml params and page frontmatter */}} {{/* ABOUTME: Wash provides a tinted gradient behind text panels on background-layout pages */}} {{/* Resolution chain: page image.wash > site params.wash > defaults */}} {{- $siteWash := .Site.Params.wash | default dict -}} {{- $pageWash := .Params.image.wash | default dict -}} {{- $opacity := 0.25 -}} {{- with $pageWash.opacity -}}{{- $opacity = . -}} {{- else -}}{{- with $siteWash.opacity -}}{{- $opacity = . -}}{{- end -}} {{- end -}} {{- $blur := "0px" -}} {{- with $pageWash.blur -}}{{- $blur = . -}} {{- else -}}{{- with $siteWash.blur -}}{{- $blur = . -}}{{- end -}} {{- end -}} {{/* Gradient edges: left top right bottom — how far the fade extends inward */}} {{- $gradientRaw := "0% 10% 0% 10%" -}} {{- with $pageWash.gradient -}}{{- $gradientRaw = . -}} {{- else -}}{{- with $siteWash.gradient -}}{{- $gradientRaw = . -}}{{- end -}} {{- end -}} {{/* Parse gradient string into four values */}} {{- $parts := split $gradientRaw " " -}} {{- $gradLeft := "0%" -}} {{- $gradTop := "10%" -}} {{- $gradRight := "0%" -}} {{- $gradBottom := "10%" -}} {{- if eq (len $parts) 4 -}} {{- $gradLeft = index $parts 0 -}} {{- $gradTop = index $parts 1 -}} {{- $gradRight = index $parts 2 -}} {{- $gradBottom = index $parts 3 -}} {{- else if eq (len $parts) 2 -}} {{/* shorthand: horizontal vertical */}} {{- $gradLeft = index $parts 0 -}} {{- $gradTop = index $parts 1 -}} {{- $gradRight = index $parts 0 -}} {{- $gradBottom = index $parts 1 -}} {{- else if eq (len $parts) 1 -}} {{/* single value: all edges */}} {{- $gradLeft = index $parts 0 -}} {{- $gradTop = index $parts 0 -}} {{- $gradRight = index $parts 0 -}} {{- $gradBottom = index $parts 0 -}} {{- end -}}
