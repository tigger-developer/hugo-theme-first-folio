{{/* ABOUTME: Generates CSS custom properties for heading prefix characters */}} {{/* ABOUTME: Controls the decorative marks before titles and headings in all contexts. - card prefix on masonry card titles - carousel prefix on carousel card titles - list prefix on list-view items (sections with list_style: list) - page prefix on the article page title (banner, hero, etc.) - sectionTitle prefix on section page titles (e.g. /blog/, /poetry/) - h1..h6 prefix on each heading level inside article body content Set any value to "" to suppress the prefix entirely (no glyph, no spacing). */}} {{- $hp := .Site.Params.headingprefix | default dict -}} {{/* Non-breaking space (U+00A0) keeps the prefix glued to the title text. */}} {{- $thinSpace := printf "%c" 0x00A0 -}} {{- $resolve := dict "card" (printf "#%s" $thinSpace) "carousel" (printf "#%s" $thinSpace) "list" "" "page" (printf "#%s" $thinSpace) "sectiontitle" (printf "#%s" $thinSpace) "h1" (printf "#%s" $thinSpace) "h2" (printf "##%s" $thinSpace) "h3" (printf "###%s" $thinSpace) "h4" (printf "####%s" $thinSpace) "h5" (printf "#####%s" $thinSpace) "h6" (printf "######%s" $thinSpace) -}} {{- $values := dict -}} {{- range $key, $default := $resolve -}} {{- $val := $default -}} {{- if isset $hp $key -}} {{- $set := index $hp $key -}} {{- if eq $set "" -}} {{- $val = "" -}} {{- else -}} {{- $val = printf "%s%s" $set $thinSpace -}} {{- end -}} {{- end -}} {{- $values = merge $values (dict $key $val) -}} {{- end -}}
