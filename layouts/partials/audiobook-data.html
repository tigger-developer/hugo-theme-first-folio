{{/* ABOUTME: Validates audiobook frontmatter and returns normalized book/chapter data. ABOUTME: Shared by the audiobook HTML page and podcast feed templates. */}} {{- $page := . -}} {{- $book := .Params.audiobook -}} {{- if not $book -}} {{- errorf "audiobook metadata requires params.audiobook on %s" .Path -}} {{- end -}} {{- if not $book.id -}} {{- errorf "audiobook metadata requires id on %s" .Path -}} {{- end -}} {{- if or (not $book.title) (not $book.description) -}} {{- errorf "audiobook metadata requires title and description on %s" .Path -}} {{- end -}} {{- $language := $book.language | default .Site.Language.Locale | default .Site.Language.Lang -}} {{- if not $language -}} {{- errorf "audiobook metadata requires language on %s" .Path -}} {{- end -}} {{- $type := $book.type | default "episodic" -}} {{- if not (in (slice "serial" "episodic") $type) -}} {{- errorf "audiobook metadata type must be serial or episodic on %s" .Path -}} {{- end -}} {{- if not (isset $book "chapters") -}} {{- errorf "audiobook metadata requires chapters on %s" .Path -}} {{- end -}} {{- $chapters := $book.chapters -}} {{- if not $chapters -}} {{- errorf "audiobook metadata requires at least one chapter on %s" .Path -}} {{- end -}} {{- $generatedRoot := index hugo.Data "first_folio_media" | default dict -}} {{- $generatedBook := index $generatedRoot $book.id | default dict -}} {{- $subscribe := $book.subscribe | default dict -}} {{- $save := $book.save | default dict -}} {{- $homescreen := $book.homescreen | default dict -}} {{- $seen := dict -}} {{- $normalized := slice -}} {{- range $index, $chapter := $chapters -}} {{- $chapterID := $chapter.id -}} {{- if not $chapterID -}} {{- errorf "audiobook chapter at index %d is missing id on %s" $index $page.Path -}} {{- end -}} {{- if index $seen $chapterID -}} {{- errorf "audiobook chapter id %s is duplicated on %s" $chapterID $page.Path -}} {{- end -}} {{- $seen = merge $seen (dict $chapterID true) -}} {{- if not $chapter.title -}} {{- errorf "audiobook chapter %s is missing title on %s" $chapterID $page.Path -}} {{- end -}} {{- if not $chapter.src -}} {{- errorf "audiobook chapter %s is missing src on %s" $chapterID $page.Path -}} {{- end -}} {{- $mediaSrc := partial "audiobook-media-src.html" (dict "page" $page "src" $chapter.src "chapterID" $chapterID) -}} {{- $generated := index $generatedBook $chapterID | default dict -}} {{- with $generated.src -}} {{- if ne . $chapter.src -}} {{- errorf "audiobook media metadata src mismatch for chapter %s on %s: generated %q does not match front matter %q" $chapterID $page.Path . $chapter.src -}} {{- end -}} {{- end -}} {{- $mimeType := or (index $chapter "mimeType") (index $chapter "mimetype") (index $generated "mimeType") (index $generated "mimetype") -}} {{- if not $mimeType -}} {{- errorf "audiobook chapter %s is missing mimeType on %s" $chapterID $page.Path -}} {{- end -}} {{- $byteLength := or (index $generated "byteLength") (index $generated "bytelength") -}} {{- $byteLength = or $byteLength (index $mediaSrc "byteLength") -}} {{- if not $byteLength -}} {{- if and (not (urls.Parse $chapter.src).Scheme) (hasPrefix $chapter.src "/") -}} {{- $staticPath := printf "static/%s" (strings.TrimPrefix "/" $chapter.src) -}} {{- if os.FileExists $staticPath -}} {{- with os.Stat $staticPath -}} {{- $byteLength = .Size -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}} {{- $byteLength = or $byteLength (index $chapter "byteLength") (index $chapter "bytelength") -}} {{- if not $byteLength -}} {{- errorf "audiobook chapter %s is missing byteLength; audiobook media metadata could not resolve byteLength on %s" $chapterID $page.Path -}} {{- end -}} {{- $date := or $chapter.date $generated.date -}} {{- if not $date -}} {{- $date = $page.Date -}} {{- if eq $type "serial" -}} {{- $date = (time.AsTime $page.Date).Add (time.ParseDuration (printf "%ds" $index)) -}} {{- end -}} {{- end -}} {{- $label := $chapter.label -}} {{- if not $label -}} {{- if eq $type "episodic" -}} {{- $label = printf "Episode %v" (or $chapter.episode (add $index 1)) -}} {{- else if and (eq $index 0) (eq (urlize $chapterID) "front-matter") -}} {{- $label = "Front Matter" -}} {{- else if eq $type "serial" -}} {{- $chapterNumber := add $index 1 -}} {{- if eq (urlize (index $chapters 0).id) "front-matter" -}} {{- $chapterNumber = $index -}} {{- end -}} {{- $label = printf "Chapter %d" $chapterNumber -}} {{- else -}} {{- $label = $chapter.title -}} {{- end -}} {{- end -}} {{- $normalized = $normalized | append (dict "id" $chapterID "title" $chapter.title "label" $label "src" $chapter.src "srcRelPermalink" $mediaSrc.rel "srcPermalink" $mediaSrc.abs "mimeType" $mimeType "byteLength" $byteLength "duration" (or $generated.duration $chapter.duration) "summary" $chapter.summary "date" $date "episode" $chapter.episode ) -}} {{- end -}} {{- return (dict "id" $book.id "title" $book.title "description" $book.description "language" $language "explicit" ($book.explicit | default false) "type" $type "author" ($book.author | default .Site.Title) "image" $book.image "subscribe" (dict "label" ($subscribe.label | default "Listen in your favourite podcast app") "prompt" ($subscribe.prompt | default "Copy this Podcast Feed Link.") "hint" ($subscribe.hint | default "In your podcast app, look for Add Link or Add Feed, then paste it there.") "footnote" ($subscribe.footnote | default "Unfortunately Apple and Android do not provide a reliable one-tap Link for private feeds.") "copied" ($subscribe.copied | default "Copied Podcast Feed Link") ) "save" (dict "label" ($save.label | default "Share or save this page") "prompt" ($save.prompt | default "Copy this page Link, or use your browser's bookmark/share button to come back later.") "copied" ($save.copied | default "Copied Page Link") ) "homescreen" (dict "enabled" ($homescreen.enabled | default true) "label" ($homescreen.label | default "Save to your Home Screen") "prompt" ($homescreen.prompt | default "Save this page to your phone so it opens like an app and is easy to find again.") "ios" ($homescreen.ios | default "On iPhone or iPad, tap Share, then choose Add to Home Screen.") "android" ($homescreen.android | default "On Android, open your browser menu, then choose Add to Home screen or Install app.") "fallback" ($homescreen.fallback | default "Use your browser menu to bookmark this page or add it to your device home screen.") ) "chapters" $normalized ) -}}
