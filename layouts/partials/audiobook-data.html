{{/* ABOUTME: Validates audiobook frontmatter and returns normalized book/chapter data. ABOUTME: Shared by the audiobook HTML page and podcast feed templates. */}} {{- $page := . -}} {{- $book := .Params.audiobook -}} {{- if not $book -}} {{- errorf "audiobook metadata requires params.audiobook on %s" .Path -}} {{- end -}} {{- if not $book.id -}} {{- errorf "audiobook metadata requires id on %s" .Path -}} {{- end -}} {{- if or (not $book.title) (not $book.description) -}} {{- errorf "audiobook metadata requires title and description on %s" .Path -}} {{- end -}} {{- $language := $book.language | default .Site.Language.Locale | default .Site.Language.Lang -}} {{- if not $language -}} {{- errorf "audiobook metadata requires language on %s" .Path -}} {{- end -}} {{- $type := $book.type | default "serial" -}} {{- if not (in (slice "serial" "episodic") $type) -}} {{- errorf "audiobook metadata type must be serial or episodic on %s" .Path -}} {{- end -}} {{- if not (isset $book "chapters") -}} {{- errorf "audiobook metadata requires chapters on %s" .Path -}} {{- end -}} {{- $chapters := $book.chapters -}} {{- if not $chapters -}} {{- errorf "audiobook metadata requires at least one chapter on %s" .Path -}} {{- end -}} {{- $generatedRoot := index hugo.Data "first_folio_media" | default dict -}} {{- $generatedBook := index $generatedRoot $book.id | default dict -}} {{- $seen := dict -}} {{- $normalized := slice -}} {{- range $index, $chapter := $chapters -}} {{- $chapterID := $chapter.id -}} {{- if not $chapterID -}} {{- errorf "audiobook chapter at index %d is missing id on %s" $index $page.Path -}} {{- end -}} {{- if index $seen $chapterID -}} {{- errorf "audiobook chapter id %s is duplicated on %s" $chapterID $page.Path -}} {{- end -}} {{- $seen = merge $seen (dict $chapterID true) -}} {{- if not $chapter.title -}} {{- errorf "audiobook chapter %s is missing title on %s" $chapterID $page.Path -}} {{- end -}} {{- if not $chapter.src -}} {{- errorf "audiobook chapter %s is missing src on %s" $chapterID $page.Path -}} {{- end -}} {{- $mediaSrc := partial "audiobook-media-src.html" (dict "page" $page "src" $chapter.src "chapterID" $chapterID) -}} {{- $generated := index $generatedBook $chapterID | default dict -}} {{- with $generated.src -}} {{- if ne . $chapter.src -}} {{- errorf "audiobook media metadata src mismatch for chapter %s on %s: generated %q does not match front matter %q" $chapterID $page.Path . $chapter.src -}} {{- end -}} {{- end -}} {{- $mimeType := or (index $chapter "mimeType") (index $chapter "mimetype") (index $generated "mimeType") (index $generated "mimetype") -}} {{- if not $mimeType -}} {{- errorf "audiobook chapter %s is missing mimeType on %s" $chapterID $page.Path -}} {{- end -}} {{- $byteLength := or (index $generated "byteLength") (index $generated "bytelength") -}} {{- $byteLength = or $byteLength (index $mediaSrc "byteLength") -}} {{- if not $byteLength -}} {{- if and (not (urls.Parse $chapter.src).Scheme) (hasPrefix $chapter.src "/") -}} {{- $staticPath := printf "static/%s" (strings.TrimPrefix "/" $chapter.src) -}} {{- if os.FileExists $staticPath -}} {{- with os.Stat $staticPath -}} {{- $byteLength = .Size -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}} {{- $byteLength = or $byteLength (index $chapter "byteLength") (index $chapter "bytelength") -}} {{- if not $byteLength -}} {{- errorf "audiobook chapter %s is missing byteLength; audiobook media metadata could not resolve byteLength on %s" $chapterID $page.Path -}} {{- end -}} {{- $date := or $chapter.date $generated.date $page.Date -}} {{- $normalized = $normalized | append (dict "id" $chapterID "title" $chapter.title "src" $chapter.src "srcRelPermalink" $mediaSrc.rel "srcPermalink" $mediaSrc.abs "mimeType" $mimeType "byteLength" $byteLength "duration" (or $generated.duration $chapter.duration) "summary" $chapter.summary "date" $date "episode" $chapter.episode ) -}} {{- end -}} {{- return (dict "id" $book.id "title" $book.title "description" $book.description "language" $language "explicit" ($book.explicit | default false) "type" $type "author" ($book.author | default .Site.Title) "image" $book.image "chapters" $normalized ) -}}
